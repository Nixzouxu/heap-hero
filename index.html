// == FUNGSI MENAMBAH DATA ==
void masukkanData(int angkaBaru) {
heap[ukuran] = angkaBaru;
int indexSekarang = ukuran;
ukuran++;
// Proses Naik Jabatan (Bubble Up)
while (indexSekarang != 0) {
int indexAyah = (indexSekarang - 1) / 2;
if (heap[indexSekarang] > heap[indexAyah]) {
tukar(&heap[indexSekarang], &heap[indexAyah]);
indexSekarang = indexAyah;
} else {
break;
}
}
}
// == FUNGSI MENGAMBIL DATA PUNCAK ==
int ambilMaksimal() {
int nilaiMaksimal = heap[0];
heap[0] = heap[ukuran - 1];
ukuran--;
perbaikiKeBawah(0);
return nilaiMaksimal;
}
// == FUNGSI TURUN JABATAN (HEAPIFY DOWN) ==

int indexTerbesar = index;
int anakKiri = (2 * index) + 1;
